shuffleArray = array => {
    const newArray = [...array];
  for (let i = newArray.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    const temp = newArray[i];
    newArray[i] = newArray[j];
    newArray[j] = temp;
  }
  return newArray;
}

function factorial(n) {
    let val = 1;
    let N = n;
    while (N > 1) {
        val = val * N;
        N--;
    }
    return val;
}

function distributions(Nblack, Nred) {
    const Npairs = (Nblack + Nred)/2
    const N_BR_max = Math.min(Nblack, Nred)
    const N_BR_min = Nred % 2 == 0 ? 0 : 1
    const N_BR_length = (N_BR_max - N_BR_min) / 2 + 1
    let ways = Array.from({length: N_BR_length}).map((x, i) => {
        const N_BR = N_BR_min + i*2;
        const N_B2 = (Nblack - N_BR)/2;
        const N_R2 = (Nred - N_BR)/2;
        return {N_B2: N_B2, N_R2: N_R2, N_BR: N_BR, N_BR_min: N_BR-0.5, N_BR_max: N_BR+0.5, ways: factorial(Npairs)/ (factorial(N_B2) * factorial(N_R2) * factorial(N_BR)) * Math.pow(2, N_BR)};
    })
    const ways_array = ways.map(x => x.ways)
    const ways_sum = ways_array.reduce((a, b) => a + b, 0)
    const ways_max = ways_array.reduce((a, b) => Math.max(a, b), 0)
    ways.forEach(x => {
        x.relWays = x.ways / ways_max;
        x.prob = x.ways / ways_sum;
    });
    return ways;
}
